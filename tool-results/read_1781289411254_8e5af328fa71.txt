   194вЖТ   194вЖТ// вФАвФАвФА Telegram HTML вЖТ rich HTML вФАвФАвФА
   195вЖТ   195вЖТfunction telegramHtmlToRich(html: string): string {
   196вЖТ   196вЖТ  return html
   197вЖТ   197вЖТ    .replace(/([^>\n])\n(?![<])/g, '$1<br/>')
   198вЖТ   198вЖТ    .replace(/\n(?!<)/g, '<br/>')
   199вЖТ   199вЖТ    .replace(/<br\s*\/>/gi, '<br/>')
   200вЖТ   200вЖТ    .replace(/<br\s*>/gi, '<br/>')
   201вЖТ   201вЖТ    .replace(/<\/p>/gi, '<br/>')
   202вЖТ   202вЖТ    .replace(/<p[^>]*>/gi, '')
   203вЖТ   203вЖТ    .replace(/<\/div>/gi, '<br/>')
   204вЖТ   204вЖТ    .replace(/<div[^>]*>/gi, '')
   205вЖТ   205вЖТ    .replace(/<\/li>/gi, '<br/>')
   206вЖТ   206вЖТ    .replace(/<li[^>]*>/gi, '<span style="display:inline-block;margin-left:8px">вАҐ</span> ')
   207вЖТ   207вЖТ    .replace(/<ul[^>]*>|<ol[^>]*>/gi, '')
   208вЖТ   208вЖТ    .replace(/<\/ul>|<\/ol>/gi, '<br/>')
   209вЖТ   209вЖТ    .replace(/<blockquote>/gi, '<blockquote>')
   210вЖТ   210вЖТ    .replace(/<\/blockquote>/gi, '</blockquote><br/>')
   211вЖТ   211вЖТ    .replace(/(<br\s*\/>){3,}/g, '<br/><br/>')
   212вЖТ   212вЖТ  return sanitizeTelegramHtml(html)
   213вЖТ   213вЖТ}

   339вЖТ   339вЖТ// вФАвФАвФА Generate 7 days (today + 6 days) in MSK (UTC+3) вФАвФАвФА
   340вЖТ   340вЖТ// The schedule system is entirely MSK-based, so the client must also use MSK
   341вЖТ   341вЖТ// to determine the current day вАФ otherwise users in other timezones see the wrong day.
   342вЖТ   342вЖТfunction getMSKWeekDays(): { date: string; label: string; shortLabel: string; isToday: boolean }[] {
   343вЖТ   343вЖТ  const days: { date: string; label: string; shortLabel: string; isToday: boolean }[] = []
   344вЖТ   344вЖТ  const dayNames = ['–Т—Б', '–Я–љ', '–Т—В', '–°—А', '–І—В', '–Я—В', '–°–±']
   345вЖТ   345вЖТ  const monthNames = ['—П–љ–≤', '—Д–µ–≤', '–Љ–∞—А', '–∞–њ—А', '–Љ–∞—П', '–Є—О–љ', '–Є—О–ї', '–∞–≤–≥', '—Б–µ–љ', '–Њ–Ї—В', '–љ–Њ—П', '–і–µ–Ї']
   346вЖТ   346вЖТ
   347вЖТ   347вЖТ  // Compute current MSK time: Date.now() is always UTC ms, just add +3h
   348вЖТ   348вЖТ  const mskNow = new Date(Date.now() + 3 * 3600000)
   349вЖТ   349вЖТ
   350вЖТ   350вЖТ  for (let i = 0; i < 7; i++) {
   351вЖТ   351вЖТ    const d = new Date(mskNow)
   352вЖТ   352вЖТ    d.setUTCDate(d.getUTCDate() + i)
   353вЖТ   353вЖТ
   354вЖТ   354вЖТ    const yyyy = d.getUTCFullYear()
   355вЖТ   355вЖТ    const mm = String(d.getUTCMonth() + 1).padStart(2, '0')
   356вЖТ   356вЖТ    const dd = String(d.getUTCDate()).padStart(2, '0')
   357вЖТ   357вЖТ    const dateStr = `${yyyy}-${mm}-${dd}`
   358вЖТ   358вЖТ
   359вЖТ   359вЖТ    const dayName = dayNames[d.getUTCDay()]
   360вЖТ   360вЖТ
   361вЖТ   361вЖТ    if (i === 0) {
   362вЖТ   362вЖТ      days.push({ date: dateStr, label: '–°–µ–≥–Њ–і–љ—П', shortLabel: '–°–µ–≥–Њ–і–љ—П', isToday: true })
   363вЖТ   363вЖТ    } else if (i === 1) {
   364вЖТ   364вЖТ      days.push({ date: dateStr, label: '–Ч–∞–≤—В—А–∞', shortLabel: '–Ч–∞–≤—В—А–∞', isToday: false })
   365вЖТ   365вЖТ    } else {
   366вЖТ   366вЖТ      days.push({
   367вЖТ   367вЖТ        date: dateStr,
   368вЖТ   368вЖТ        label: `${dayName}, ${d.getUTCDate()} ${monthNames[d.getUTCMonth()]}`,
   369вЖТ   369вЖТ        shortLabel: dayName,
   370вЖТ   370вЖТ        isToday: false,
   371вЖТ   371вЖТ      })
   372вЖТ   372вЖТ    }
   373вЖТ   373вЖТ  }
   374вЖТ   374вЖТ  return days
   375вЖТ   375вЖТ}
