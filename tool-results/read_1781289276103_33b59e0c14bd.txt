    71→    def load_config(self) -> None:
    72→        """Загружает конфигурацию из JSON-файла."""
    73→        config_file = Path(self.config_path).resolve()
    74→        if not config_file.exists():
    75→            print(f"[BOT] Файл конфигурации не найден: {config_file}")
    76→            print(f"[BOT] Текущая директория: {os.getcwd()}")
    77→            logger.error("Файл конфигурации не найден: %s", config_file)
    78→            return
    79→
    80→        try:
    81→            with open(config_file, "r", encoding="utf-8") as f:
    82→                self.config = json.load(f)
    83→            logger.info("Конфигурация загружена из %s (%d ключей)", config_file, len(self.config))
    84→
    85→            # Web Panel URL и API ключ
    86→            self.web_panel_url = self.config.get("web_panel_url", "")
    87→            self.web_panel_api_key = self.config.get("web_panel_api_key", "")
    88→            self.notify_chat_id = str(self.config.get("telegram_notify_chat_id", ""))
    89→            self.moderation_notifications = self.config.get("moderation_notifications", True)
    90→            if self.web_panel_url:
    91→                logger.info("Веб-панель: %s", self.web_panel_url)
    92→            if self.notify_chat_id:
    93→                logger.info("Уведомления будут отправляться в чат %s", self.notify_chat_id)
    94→        except Exception as e:
    95→            print(f"[BOT] Ошибка чтения конфига: {e}")
    96→            logger.error("Ошибка чтения конфига: %s", e)

   364→    async def _get_admin_chat_ids(self) -> list[int]:
   365→        """Возвращает список chat_id всех зарегистрированных пользователей бота."""
   366→        if not self.db:
   367→            return []
   368→        try:
   369→            users = await self.db.get_all_subscribers()
   370→            return [u["user_id"] for u in users if u.get("user_id")]
   371→        except Exception as exc:
   372→            logger.warning("Не удалось получить список пользователей: %s", exc)
   373→            return []

   375→    async def _notify_moderation(
   376→        self, title: str, news_type: str, priority: str, source: str
   377→    ) -> None:
   378→        """Отправляет уведомление о новой новости на модерацию всем админам бота."""
   379→        if not self.moderation_notifications or not self.publisher:
   380→            return
   381→
   382→        # Получаем список chat_id из базы бота
   383→        admin_ids = await self._get_admin_chat_ids()
   384→        if not admin_ids:
   385→            # Фоллбэк: если в базе никого нет — берём из config
   386→            if self.notify_chat_id:
   387→                admin_ids = [int(self.notify_chat_id)]
   388→            else:
   389→                return
   390→
   391→        type_icons = {
   392→            "update": "🎮", "wipe": "🔄", "patch": "🔧", "event": "📅",
   393→            "maintenance": "🛠️", "bug": "🐛", "mod": "🔧", "guide": "📖",
   394→            "story": "📖", "tip": "💡", "discussion": "💬", "meme": "😂",
   395→            "content": "📷", "other": "📰",
   396→        }
   397→        priority_labels = {"high": "🔴 Высокий", "medium": "🟡 Средний", "low": "🟢 Низкий"}
   398→        icon = type_icons.get(news_type, "📰")
   399→        prio = priority_labels.get(priority, priority)
   400→
   401→        # Экранируем динамические данные для Telegram HTML
   402→        safe_title = html_module.escape(title[:80])
   403→        safe_source = html_module.escape(source)
   404→        safe_type = html_module.escape(news_type)
   405→        safe_prio = html_module.escape(prio)
   406→
   407→        text = (
   408→            f"{icon} <b>Новость на модерации</b>\n"
   409→            f"━━━━━━━━━━━━━━━━━\n"
   410→            f"📌 <b>Тип:</b> {safe_type}\n"
   411→            f"⚡ <b>Приоритет:</b> {safe_prio}\n"
   412→            f"📡 <b>Источник:</b> {safe_source}\n\n"
   413→            f"💬 <i>{safe_title}</i>\n\n"
   414→            f"🔗 <a href=\"{self.web_panel_url}/dashboard/moderation\">Открыть модерацию</a>"
   415→        )
   416→
   417→        for chat_id in admin_ids:
   418→            try:
   419→                await self.publisher.bot.send_message(
   420→                    chat_id=chat_id,
   421→                    text=text,
   422→                    parse_mode=ParseMode.HTML,
   423→                )
   424→                logger.debug("Уведомление о модерации отправлено: chat_id=%d", chat_id)
   425→            except Exception as exc:
   426→                logger.warning("Не удалось отправить уведомление chat_id=%d: %s", chat_id, exc)
   427→
   428→    async def _task_check_pending_moderation(self) -> None:
   429→        """Периодически проверяет веб-панель на наличие ожидающих модерацию новостей."""
   430→        if not self.moderation_notifications or not self.web_panel_url or not HAS_WEB_PANEL:
   431→            return
   432→        if not self.publisher:
   433→            return
   434→        try:
   435→            status = await get_moderation_status(
   436→                self.web_panel_url,
   437→                bot_api_key=self.web_panel_api_key,
   438→            )
   439→            pending = status.get("pending", 0)
   440→
   441→            if pending > 0 and pending > self._last_pending_count:
   442→                # Отправляем отдельное уведомление по каждой новой новости
   443→                items = status.get("items", [])
   444→                new_items = items[:pending - self._last_pending_count]
   445→                admin_ids = await self._get_admin_chat_ids()
   446→                for item in new_items:
   447→                    title = item.get("summary", item.get("content", "Без описания"))[:80]
   448→                    news_type = item.get("newsType", item.get("news_type", "other"))
   449→                    priority = item.get("priority", "medium")
   450→                    source = item.get("source", item.get("source_type", "unknown"))
   451→                    await self._notify_moderation(title, news_type, priority, source)
   452→            self._last_pending_count = pending
   453→        except Exception as exc:
   454→            logger.debug("Ошибка проверки модерации: %s", exc)

   456→    async def _task_analyze_messages(self) -> None:
   457→        """AI-анализ необработанных сообщений + дедупликация."""
   458→        if not self.ai_analyzer or not self.db:
   459→            return
   460→        try:
   461→            # Берём максимум 3 за раз — чтобы не спамить пачками
   462→            messages = await self.db.get_unprocessed_messages(limit=3)
   463→            if not messages:
   464→                return
   465→
   466→            processed_count = 0
   467→            for msg in messages:
   468→                msg_id = msg["id"]
   469→                text = msg.get("text", "")
   470→
   471→                # Дедупликация
   472→                if self.deduplicator:
   473→                    try:
   474→                        images = json.loads(msg.get("images", "[]"))
   475→                    except (json.JSONDecodeError, TypeError):
   476→                        images = []
   477→
   478→                    # Для Reddit — более строгая дедупликация (порог 0.65 вместо 0.85)
   479→                    old_threshold = self.deduplicator.similarity_threshold
   480→                    if msg.get("source_type") == "reddit":
   481→                        self.deduplicator.similarity_threshold = 0.65
   482→
   483→                    duplicate_of = await self.deduplicator.is_duplicate(
   484→                        msg_id, text, images
   485→                    )
   486→
   487→                    # Возвращаем порог обратно
   488→                    self.deduplicator.similarity_threshold = old_threshold
   489→
   490→                    if duplicate_of:
   491→                        await self.deduplicator.mark_as_duplicate(
   492→                            duplicate_of, msg_id
   493→                        )
   494→                        continue
   495→
   496→                # AI-анализ
   497→                author = msg.get("author", "") or msg.get("server_name", "")
   498→                source_type = msg.get("source_type", "")
   499→
   500→                if source_type == "reddit":
   501→                    subreddit = msg.get("channel_name", "")
   502→                    result = await self.ai_analyzer.analyze_reddit(text, author=author, subreddit=subreddit)
   503→                else:
   504→                    result = await self.ai_analyzer.analyze(text, author=author)
   505→
   506→                if result:
   507→                    # Всегда сохраняем с should_publish=False в локальной БД,
   508→                    # чтобы авто-публикатор _task_publish_pending не публиковал
   509→                    # новости без модерации. Публикация — ТОЛЬКО после одобрения
   510→                    # через веб-панель.
   511→                    await self.db.save_processed(
   512→                        message_id=msg_id,
   513→                        news_type=result["news_type"],
   514→                        priority=result["priority"],
   515→                        should_publish=False,
   516→                        summary=result["summary"],
   517→                        server_name=result.get("server_name", ""),
   518→                        formatted_post=result.get("formatted_post", ""),
   519→                    )
   520→
   521→                    logger.info(
   522→                        "Сообщение #%d отправлено на модерацию (type=%s, priority=%s)",
   523→                        msg_id, result["news_type"], result["priority"],
   524→                    )
   525→
   526→                    # Отправить на веб-панель для модерацию
   527→                    if HAS_WEB_PANEL:
   528→                        if not self.web_panel_url:
   529→                            logger.warning("Веб-панель: URL не настроен в config.json (web_panel_url)")
   530→                        else:
   531→                            try:
   532→                                success = await send_to_web_panel(
   533→                                    news_data={
   534→                                        "externalId": str(msg_id),
   535→                                        "serverName": result.get("server_name", "") or author,
   536→                                        "content": text,
   537→                                        "summary": result.get("summary", ""),
   538→                                        "formattedPost": result.get("formatted_post", ""),
   539→                                        "newsType": result.get("news_type", "other"),
   540→                                        "priority": result.get("priority", "low"),
   541→                                        "images": json.loads(msg.get("images", "[]")) if msg.get("images") else [],
   542→                                    },
   543→                                    web_app_url=self.web_panel_url,
   544→                                    bot_api_key=self.web_panel_api_key or None,
   545→                                )
   546→                                if success:
   547→                                    await self._notify_moderation(
   548→                                        title=result.get("summary", "")[:80],
   549→                                        news_type=result.get("news_type", "other"),
   550→                                        priority=result.get("priority", "low"),
   551→                                        source=result.get("server_name", "") or author,
   552→                                    )
   553→                                elif not success:
   554→                                    logger.error("Веб-панель: не удалось отправить новость #%d", msg_id)
   555→                            except Exception as web_err:
   556→                                logger.error("Веб-панель: исключение при отправке #%d: %s", msg_id, web_err)
   557→                    else:
   558→                        logger.warning("Веб-панель: модуль web_app_integration не загружен")
   559→                else:
   560→                    await self.db.save_processed(
   561→                        message_id=msg_id,
   562→                        news_type="other",
   563→                        priority="low",
   564→                        should_publish=False,
   565→                        summary="Ошибка анализа",
   566→                    )
   567→                    # Всё равно отправляем на веб-панель как черновик
